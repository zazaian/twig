  def friends(action=nil)
    if @friends
      case action
      when :update
        @friends = Twig::Timeline::Friends.new(@client)
      when nil
        @friends
      else
        raise ArgumentError, "#{action} is not a valid argument for @active."
      end
    else
      @friends = Twig::Timeline::Friends.new(@client)
    end
  end
