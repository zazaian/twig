  def users(action=nil)
    if @users
      case action
      when :clear
        @users = Twig::Users.new(@client)
      when nil
        @users
      else
        raise ArgumentError, "#{action} is not a valid argument for @users."
      end
    else
      @users = Twig::Users.new(@client)
    end
  end
