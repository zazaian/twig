# File lib/twig/message/message.rb, line 16
  def inbox(action=nil)
    if @inbox
      case action
      when :destroy, :empty
        @inbox = Twig::Inbox.new(@client)
      when nil
        @inbox
      else
        raise ArgumentError, "#{action} is not a valid argument for @active."
      end
    else
      @inbox = Twig::Inbox.new(@client)
    end
  end
