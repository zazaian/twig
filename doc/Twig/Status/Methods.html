  def status(action=nil)
    if @status
      case action
      when :create
        @status = Twig::Status.new(@client)
      when :update
        @status.update
      when nil
        @status
      else
        raise ArgumentError, "#{action} is not a valid argument for @status."
      end
    else
      @status = Twig::Status.new(@client)
    end
  end
